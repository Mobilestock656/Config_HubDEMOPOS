// ฟังก์ชันสร้างการ์ดระบบ function createCard(key, data) { const card = document.createElement('div'); card.className = `card ${!data.active ? 'disabled' : ''}`; let clickAction = ""; if (data.active) { // บังคับให้ทุกบริการเปิดแบบหน้าจอซ้อน Modal ทั้งหมดเพื่อความลื่นไหลสม่ำเสมอ clickAction = `onclick="openAppModal('${data.url}', '${data.title}')"`; } else { clickAction = `onclick="alert('${data.errorMsg}')"`; } card.innerHTML = `
${!data.active ? 'Maintenance' : data.badge}
${data.icon}

${data.title}

${data.desc}

${data.active ? 'เปิดเข้าใช้งานระบบ' : 'กำลังปรับปรุงระบบ'}
`; return card; } // ฟังก์ชันเปิดหน้าต่างย่อยซ้อนระบบหลัก function openAppModal(url, title) { const overlay = document.getElementById('appModalOverlay'); const win = document.getElementById('appModalWindow'); overlay.className = "app-modal-overlay show"; win.className = "app-modal-window"; document.getElementById('appModalTitle').innerText = title; document.getElementById('appModalIframe').src = url; // ตรวจสอบอัตโนมัติ: ถ้าระบบที่เปิดเป็นคิดเงินหน้าร้าน (pos) หรือคลังสินค้า (stock) ให้สั่งขยายเต็มจอใหญ่เต็มตาให้ทันที if (url.toLowerCase().includes('pos') || url.toLowerCase().includes('stock')) { win.classList.add('fullscreen'); } document.body.style.position = 'fixed'; document.body.style.width = '100%'; }
